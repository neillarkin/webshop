54class PrimeCurve(): 

55 """ 

56 Elliptic curve over a prime field. Characteristic two field curves are not 

57 supported. 

58 """ 

59 

60 def __init__(self, p, a, b): 

61 """ 

62 The curve of points satisfying y^2 = x^3 + a*x + b (mod p) 

63 

64 :param p: 

65 The prime number as an integer 

66 

67 :param a: 

68 The component a as an integer 

69 

70 :param b: 

71 The component b as an integer 

72 """ 

73 

74 self.p = p 

75 self.a = a 

76 self.b = b 

77 

78 def contains(self, point): 

79 """ 

80 :param point: 

81 A Point object 

82 

83 :return: 

84 Boolean if the point is on this curve 

85 """ 

86 

87 y2 = point.y * point.y 

88 x3 = point.x * point.x * point.x 

89 return (y2 - (x3 + self.a * point.x + self.b)) % self.p == 0

92class PrimePoint(): 

93 """ 

94 A point on a prime-field elliptic curve 

95 """ 

96 

97 def __init__(self, curve, x, y, order=None): 

98 """ 

99 :param curve: 

100 A PrimeCurve object 

101 

102 :param x: 

103 The x coordinate of the point as an integer 

104 

105 :param y: 

106 The y coordinate of the point as an integer 

107 

108 :param order: 

109 The order of the point, as an integer - optional 

110 """ 

111 

112 self.curve = curve 

113 self.x = x 

114 self.y = y 

115 self.order = order 

116 

117 # self.curve is allowed to be None only for INFINITY: 

118 if self.curve: 

119 if not self.curve.contains(self): 

120 raise ValueError('Invalid EC point') 

121 

122 if self.order: 

123 if self * self.order != INFINITY: 

124 raise ValueError('Invalid EC point') 

125 

126 def __cmp__(self, other): 

127 """ 

128 :param other: 

129 A PrimePoint object 

130 

131 :return: 

132 0 if identical, 1 otherwise 

133 """ 

134 if self.curve == other.curve and self.x == other.x and self.y == other.y: 

135 return 0 

136 else: 

137 return 1 

138 

139 def __add__(self, other): 

140 """ 

141 :param other: 

142 A PrimePoint object 

143 

144 :return: 

145 A PrimePoint object 

146 """ 

147 

148 # X9.62 B.3: 

149 

150 if other == INFINITY: 

151 return self 

152 if self == INFINITY: 

153 return other 

154 assert self.curve == other.curve 

155 if self.x == other.x: 

156 if (self.y + other.y) % self.curve.p == 0: 

157 return INFINITY 

158 else: 

159 return self.double() 

160 

161 p = self.curve.p 

162 

163 l_ = ((other.y - self.y) * inverse_mod(other.x - self.x, p)) % p 

164 

165 x3 = (l_ * l_ - self.x - other.x) % p 

166 y3 = (l_ * (self.x - x3) - self.y) % p 

167 

168 return PrimePoint(self.curve, x3, y3) 

169 

170 def __mul__(self, other): 

171 """ 

172 :param other: 

173 An integer to multiple the Point by 

174 

175 :return: 

176 A PrimePoint object 

177 """ 

178 

179 def leftmost_bit(x): 

180 assert x > 0 

181 result = 1 

182 while result <= x: 

183 result = 2 * result 

184 return result // 2 

185 

186 e = other 

187 if self.order: 

188 e = e % self.order 

189 if e == 0: 

190 return INFINITY 

191 if self == INFINITY: 

192 return INFINITY 

193 assert e > 0 

194 

195 # From X9.62 D.3.2: 

196 

197 e3 = 3 * e 

198 negative_self = PrimePoint(self.curve, self.x, -self.y, self.order) 

199 i = leftmost_bit(e3) // 2 

200 result = self 

201 # print "Multiplying %s by %d (e3 = %d):" % ( self, other, e3 ) 

202 while i > 1: 

203 result = result.double() 

204 if (e3 & i) != 0 and (e & i) == 0: 

205 result = result + self 

206 if (e3 & i) == 0 and (e & i) != 0: 

207 result = result + negative_self 

208 # print ". . . i = %d, result = %s" % ( i, result ) 

209 i = i // 2 

210 

211 return result 

212 

213 def __rmul__(self, other): 

214 """ 

215 :param other: 

216 An integer to multiple the Point by 

217 

218 :return: 

219 A PrimePoint object 

220 """ 

221 

222 return self * other 

223 

224 def double(self): 

225 """ 

226 :return: 

227 A PrimePoint object that is twice this point 

228 """ 

229 

230 # X9.62 B.3: 

231 

232 p = self.curve.p 

233 a = self.curve.a 

234 

235 l_ = ((3 * self.x * self.x + a) * inverse_mod(2 * self.y, p)) % p 

236 

237 x3 = (l_ * l_ - 2 * self.x) % p 

238 y3 = (l_ * (self.x - x3) - self.y) % p 

239 

240 return PrimePoint(self.curve, x3, y3)
